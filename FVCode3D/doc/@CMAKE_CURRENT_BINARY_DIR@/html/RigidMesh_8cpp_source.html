625 Real Rigid_Mesh::Cell::orientationFacet(const Facet & fac) const throw()
626 {
627  bool found(false);
628  UInt facetId = fac.getId();
629 
630  // control that facetId is a facet of this cell
631  for( std::vector<UInt>::const_iterator it = M_facetsIds.begin(); it != M_facetsIds.end(); ++it )
632  {
633  if( *it == facetId )
634  found = true;
635  }
636 
637  // check if the facet exists in this cell
638  if( !found )
639  {
640  std::stringstream error;
641  error << "Error: the facet " << facetId << " is not in the cell.";
642  throw std::runtime_error(error.str());
643  }
644 
645  const Point3D facetNormal = fac.getUnsignedNormal();
646  const Point3D g = fac.getCentroid() - Rigid_Mesh::Cell::M_centroid;
647 
648  return ( (dotProduct(g,facetNormal)>=0.) ? 1.0 : -1.0 );
649 }
