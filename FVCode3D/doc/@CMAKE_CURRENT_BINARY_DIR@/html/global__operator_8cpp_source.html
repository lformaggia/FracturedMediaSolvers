394 void FluxOperator::assemble()
395 {
396  auto & T = *M_matrix;
397  for (auto& facet_it : M_mesh.getFractureFacetsIdsVector())
398  {
399  const UInt Id_F = facet_it.getFractureId();
400  // Assemble -T matrix taking into account fractures intersections with the "star-delta" transformation
401  for (auto juncture_it : facet_it.getFractureNeighbors())
402  {
403  std::vector<Real> alphas;
404  const Real alphaF = findFracturesAlpha (juncture_it.first, facet_it.getFractureId());
405 
406  for (auto neighbors_it : juncture_it.second)
407  alphas.emplace_back(findFracturesAlpha (juncture_it.first, neighbors_it));
408 
409  Real a_sum = alphaF;
410  auto sum_maker = [&a_sum](Real item){a_sum += item;};
411  std::for_each(alphas.begin(), alphas.end(), sum_maker);
412 
413  for (UInt counter = 0; counter < alphas.size(); ++counter)
414  {
415 
416  const Real QFf = alphaF * alphas[counter] * M_mesh.getPropertiesMap().getMobility() / a_sum;
417 
418  T.coeffRef(Id_F, Id_F) -= QFf;
419 
420  T.insert(Id_F,juncture_it.second[counter]) = QFf;

810 void FractureBuilder::assemble_Monolithic()
811 {
812  for (auto& facet_it : M_mesh.getFractureFacetsIdsVector())
813  {
814  auto& F_permeability = M_mesh.getPropertiesMap().getProperties(facet_it.getZoneCode()).M_permeability;
815  const Real Kfn = F_permeability->operator()(0,0);
816  const Real F_aperture = M_mesh.getPropertiesMap().getProperties(facet_it.getZoneCode()).M_aperture;
817  const Real eta = F_aperture / Kfn;
818  const Real xsi0 = (2. * xsi - 1.) / 4.;
819 
820  const UInt numFacetsTot = M.cols();
821  const UInt numCells = M_mesh.getCellsVector().size();
822 
823  const UInt Id_plus = facet_it.getId();
824  const UInt Id_F = facet_it.getFractureId();
825  const UInt Id_minus = M_mesh.getFacetsVector().size() + Id_F;
826  const Real facetMeasure = facet_it.getSize();
827 
828  // Coupling terms on M
829  S.coeffRef(Id_plus,Id_plus) += eta*facetMeasure/4. + eta*xsi0*facetMeasure;
830  S.insert(Id_plus,Id_minus) = eta*facetMeasure/4. - eta*xsi0*facetMeasure;
831  S.coeffRef(Id_minus,Id_minus) += eta*facetMeasure/4. + eta*xsi0*facetMeasure;
832  S.insert(Id_minus,Id_plus) = eta*facetMeasure/4. - eta*xsi0*facetMeasure;
833 
834  // Coupling matrix Ct and C
835  S.insert(Id_plus, numFacetsTot+numCells + Id_F) = facetMeasure;
836  S.insert(Id_minus, numFacetsTot+numCells + Id_F) = -facetMeasure;
837  S.insert(numFacetsTot+numCells + Id_F, Id_plus) = facetMeasure;
838  S.insert(numFacetsTot+numCells + Id_F, Id_minus) = -facetMeasure;
839 
840  // Assemble -T matrix taking into account fractures intersections with the "star-delta" transformation
841  for (auto juncture_it : facet_it.getFractureNeighbors())
842  {
843  std::vector<Real> alphas;
844  const Real alphaF = FO.findFracturesAlpha (juncture_it.first, facet_it.getFractureId());
845 
846  for (auto neighbors_it : juncture_it.second)
847  alphas.emplace_back(FO.findFracturesAlpha (juncture_it.first, neighbors_it));
848 
849  Real a_sum = alphaF;
850  auto sum_maker = [&a_sum](Real item){a_sum += item;};
851  std::for_each(alphas.begin(), alphas.end(), sum_maker);
852 
853  for (UInt counter = 0; counter < alphas.size(); ++counter)
854  {
855  const Real QFf = alphaF * alphas[counter] * M_mesh.getPropertiesMap().getMobility() / a_sum;
856 
857  S.coeffRef( numFacetsTot+numCells + Id_F, numFacetsTot+numCells + Id_F ) -= QFf;
858 
859  S.insert( numFacetsTot+numCells + Id_F, numFacetsTot+numCells + juncture_it.second[counter]) = QFf;
860 
861  }
862  }
863  }
